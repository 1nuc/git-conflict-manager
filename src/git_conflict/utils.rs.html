1//!Define wrapper utils for efficient operation
2//! This module defines the utility functions required to ensure the smoothness of the conflict
3//! resolution operation, for example the Index and Repository struct in lib git2 crate does not
4//! implement the Copy and Clone traits. Those Traits are necessary to share the parameters across
5//! different functions. Being able to mutually mutate the Index and Repository enables efficient
6//! operation and avoid redundant code. Another major factor is the fact that those structs are
7//! extensivly used, so the usage of Reference counter is necessary that provides mutliple versions
8//! of one share instance in the memory, when combined with the ref cell, it allows for mutual
9//! mutation.
10use git2::{Error, Index, Oid, Repository, build::CheckoutBuilder};
11use std::{cell::RefCell, rc::Rc};
12
13use crate::Utils;
14
15/// As we are not using threads we are using rc not arc where the main difference between them is
16/// that one is signle-threaded that will throw an error if shared with multiple threads which is
17/// somehting arc can handle very well.
18///
19///wrapper for the Index of the lib git crate
20pub struct NucIndex(pub Rc<RefCell<Index>>);
21/// The index in git2 is not clonable by default
22/// the below implementation is to make it clonable
23impl Clone for NucIndex {
24    fn clone(&self) -> Self {
25        Self(Rc::clone(&self.0))
26    }
27}

38impl NucRepository {
39    ///The function is used for debugging purposes
40    ///its always been a curiosity for the developer to know exactly which files the branch's index is
41    ///containing.
42    ///The below method achieves that :).
43    pub fn print_index_contents(&self, index: &Index) {
44        for entry in index.iter() {
45            let path = String::from_utf8(entry.path).expect("unable to find path");
46            if let Ok(obj) = self.0.borrow().find_object(entry.id, None) {
47                if let Some(blob) = obj.as_blob() {
48                    let content = String::from_utf8(blob.content().to_vec())
49                        .expect("unable to fetch the content");
50                    dbg!("{:?}", path);
51                    dbg!("{:?}", content);
52                } else {
53                    dbg!("No content to display");
54                }
55            } else {
56                dbg!("No object with that entry");
57            }
58        }
59    }
60}

69impl Utils for Repository{

71    /// find the ancestor commits and trees
72    /// This function returns the ancestor OID that is required to fetch the commit and tree
73    /// OID is a hashed a reference to the file in the disk memory 
74    fn find_ancesistor(&self, other_branch: &str) -> Result<Oid, Error> {
75        let head_commits = self.head().unwrap().peel_to_commit().unwrap();
76        let other_branch_commits =self
77            .find_branch(other_branch, git2::BranchType::Local)
78            .unwrap()
79            .into_reference()
80            .peel_to_commit()
81            .expect("unable to fetch the commit");
82         self.merge_base(head_commits.id(), other_branch_commits.id())
83    }
