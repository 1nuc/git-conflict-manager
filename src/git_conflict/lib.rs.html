1//! Git Conflict Manager
2//! The library is intended to solve the recurring conflict issues in git
3//! The crate offers four main strategies to solve the conflict problem which are:
4//! 1. Accept local: it will accept the local changes and ignore the foreign branch's changes.
5//! 2. Accept Foreign: it will accept the foreign branch's changes and ignore the local changes.
6//! 3. combination: it will combine both branch's changes and remove the markers (straightforward).
7//!    All of which methodologies can cause a complex, diverged or broken trees in the terminal
8//! 4. Trees in git are essentials as they draw the chronological modification made on a Repository
9//!    Then the commit is the stamp that provides extra more details such as the user who made the
10//!    commit, time of when the commit was made and signature.
11//!    Due to that, another advanced method targeted for experienced developers was introduced which
12//!    is Tree Merge.
13//!    Tree Merge merges the tree for either (head, foreign) branches with the past tree or the
14//!    ancestor.
15//!    The user get to chose the base tree (head, foreign), he also get to chose whether the
16//!    interference of the past (parent) should be enabled or not.
17//!    if parent interference is enabled then the current tree selected by the user is directly merged
18//!    to the ancestor of the conflicted branches, resulting in a smooth straightforward tree.
19//!    The method is super useful for expert developers as they don't need to worry about bucking up
20//!    a specific branch if a conflict occurs, they simply chose parent interference and the branch
21//!    will be merged with the changes of direct ancestor.

54/// the manual control trait that inherits the Actions trait defines the a manual commiting action.
55pub trait ManualControl: Actions {
56    fn perform_manual_commit(&mut self, version: bool) -> bool {
57        let msg = format!(
58            "Resolve Conflict: Merge {} branch into {} branch",
59            self.branches().src_branch,
60            self.branches().dest_branch
61        );
62        // get the heads commits
63        // retreive the commits of "ours" branch and theres
64        let ours_parents_commits = self
65            .repo()
66            .head()
67            .expect("unable to find the head branch")
68            .peel_to_commit()
69            .expect("error peeling to commit in ours version")
70            .id();
71        let theirs_parents_commits = self
72            .repo()
73            .find_reference("MERGE_HEAD")
74            .expect("unable to find the second theirs reference")
75            .peel_to_commit()
76            .expect("error in peeling to a commit in theirs version")
77            .id();
78        // retreive the commits of "theirs" branch
79
80        let parent_commits: &[Oid] = match version {
81                true => &[ours_parents_commits],
82                false => &[ours_parents_commits,theirs_parents_commits],
83            };
84
85        self.commit(parent_commits, msg)
86    }
87}
88/// A trait that contains the necessary action required by all methodologies to resolve a conflict
89pub trait Actions {
90    //Return the index
91    fn branches(&self) -> Branches;
92
93    fn index(&self) -> RefMut<'_, Index>;
94
95    //Return the repo
96    fn repo(&self) -> RefMut<'_, Repository>;
97
98    //staging changes
99    //this function has an embedding implementation
100    fn staging(&mut self, files: Vec<String>) {
101        let mut index = self.index();
102        let _ = files
103            .iter()
104            .map(|x| {
105                let path = Path::new(x);
106                index
107                    .add_path(path)
108                    .expect("Error adding the file to the staging area");
109            })
110            .collect::<Vec<_>>();
111        index
112            .write()
113            .expect("unable to save the staged changes to memory");
114    }
115
116    ///Making a commit
117    ///this function has an embedding implementation
118    #[allow(unused_must_use)]
119    #[allow(non_snake_case)]
120    fn commit(&mut self, parent_commits: &[Oid], msg: String) -> bool {
121        let repo = self.repo();
122        let tree = repo.find_tree(self.index().write_tree().unwrap()).unwrap();
123        // Grabbing the details of the commit
124        let signature = repo.signature().unwrap().to_owned();
125        let config = repo
126            .config()
127            .expect("Error in retreiving the configuration")
128            .snapshot()
129            .expect("error in creating a snapshot");
130        // Considering the actual system time for establishing a commit
131        let now = SystemTime::now()
132            .duration_since(UNIX_EPOCH)
133            .unwrap()
134            .as_secs() as i64;
135
136        // Grabbing the author's details of the commit
137        let user_name = config.get_str("user.name").unwrap();
138        let email = config.get_str("user.email").unwrap();
139        let time = Time::new(now, 0);
140        let author = Signature::new(user_name, email, &time).expect("unable to create a borrow");
141        let p_cm = parent_commits
142            .iter()
143            .map(|x| repo.find_commit(*x).unwrap())
144            .collect::<Vec<Commit>>();
145        let p_cm = p_cm.iter().collect::<Vec<&Commit>>();
146        let p_commits = p_cm.as_slice();
147        //rust git2 doesn't automatically clean up the conflict the conflict must be deleted
148
149        match repo.commit(Some("HEAD"), &author, &signature, &msg, &tree, p_commits) {
150            Ok(_val) => {
151                //after making the commit git must know that the commit is clearing the conflict
152                //therefore, MERGE_HEAD file must be deleted to indicate the success of the merge
153                let merge_head_path = repo.path().join("MERGE_HEAD"); // I believe this is the
154                // line where the error
155                // stems
156                //repo.path outputs the content of the .git directory
157                //join "MERGE_HEAD" finds the file that starts with MERGE HEAD
158                if merge_head_path.exists() {
159                    remove_file(merge_head_path).expect("unable to remove the file");
160                    //deleting the merge conflict if the commit didn't auto delete
161                }
162                true
163            }
164            Error=>{
165                println!("{:?}", Error);
166                false
167            }
168        }
169    }
170
171    ///checking whether a conflict exists or not
172    fn does_conflict_exists(&self) -> bool {
173        self.index().has_conflicts()
174    }
175
176    ///return the file with a specific conditions, eg., Conflicted
177    fn return_conflicted_files(&self, condition: Status) -> Option<Vec<String>> {
178        let repo = self.repo();
179        let mut options = StatusOptions::new();
180        options
181            .include_untracked(false)
182            .recurse_untracked_dirs(false);
183        let status = repo.statuses(Some(&mut options)).unwrap();
184        let mut list_of_conflicted_files = Vec::new();
185        for i in status.iter() {
186            if i.status().contains(condition)
187                && let Some(path) = i.path()
188            {
189                list_of_conflicted_files.push(path.to_owned());
190            }
191        }
192        Some(list_of_conflicted_files)
193    }
